class Solution {
public:
    int search(vector<int>& nums, int target) {


        int l = 0;
        int h = nums.size() - 1;

        while(l <= h)
        //闭区间
        {
            int mid = l + (h - l )/2;
            // std::cout << mid << std::endl;
            if(nums[mid] > nums[l])
            //左侧有序
            {
                if(nums[mid] > target && nums[l] <= target)
                //判断左侧有序范围是否符合要求
                {
                    h = mid - 1;
                }else if(nums[mid] == target)
                {
                    return mid;
                }else
                {
                    l = mid + 1;
                }
            }else if(nums[mid] == nums[l])
            {
                if(nums[mid] == target)
                {
                    return mid;
                }else
                {
                    l = mid +1;
                }
            }else if(nums[mid] < nums[h])
            //右侧有序
            {
                if(nums[mid] < target && target <= nums[h])
                 //判断右侧有序范围是否符合要求
                {
                    l = mid + 1;
                }else if(nums[mid] == target)
                {
                    return mid;
                }else
                {
                    h = mid - 1;
                }
            }else if(nums[mid] == nums[h])
            {
                if(nums[mid] == target)
                {
                    return mid;
                }else
                {
                    h = mid - 1;
                }
            }

        }


        return -1;

    }
};
