class Solution {
private:
    int eatingSpeed(vector<int>& piles, int h)
    {
        if(h == 0)
        {
            // return 0;
            h = 1;
        }
        int k = 0;
        for(int i = 0; i < piles.size(); ++i)
        {
            // std::cout << h << std::endl;
            if(piles[i] % h == 0)
            {
                k += piles[i]/h;
            }else
            {
                k += piles[i]/h + 1;
            }
        }

        return k;
    }
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        if(h == 0)
        {
            return -1;
        }
        int max = 0;
        for_each(piles.begin(),piles.end(),[&](auto item){
            if(max < item)
            {
                max = item;
            }
        });

        int low = 1;
        int high = max;

        while(low <= high)
        {
            int mid = low + (high - low) / 2;
            // std::cout << low << " " << high << " " << mid << std::endl;
            if(eatingSpeed(piles,mid) <= h)
            {
                if(mid == 1)
                {
                    return mid;
                }
                high = mid - 1;
            }else{
                low = mid + 1;
            }
        }

        if(eatingSpeed(piles, high) <= h)
        {
            return high;
        }
        return high + 1;

    }
};
